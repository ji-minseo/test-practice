class Solution {
    public static int solution(int[] stones, int k) {
        int answer = 0;
        int niniz = 0;
        
        while (true) {
            for(int stone : stones) {
                stone--;
                if(stone == 0) {//0 나오기 시작! 여기부터 돌면서 0이 아닐때까지 쭉 cnt++
                    int cnt = 0;
                    for (int i = stone; i < stones.length; i++) {
                        if(cnt > k) {
                            return answer;
                        }
                        
                        if(stones[i] == 0) {
                            cnt++;
                            System.out.println(cnt);
                        }
                        
                    }
                }
                else {
                    answer++;
                }

            }
        }
