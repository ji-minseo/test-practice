# 두 정수 사이의 합
```
class Solution {
  public long solution(int a, int b) {
      long answer = 0;
      if(a == b) {
          return a;
      }
      else if(a>b) {
          int temp = a;
          a = b;
          b = temp;
          // a가 b보다 크면 b와 a 자리바꾸기
      }
      
            
      for (int i = a; i <= b; i++) {
          answer+=i;
          // System.out.println(answer);
      }
      return answer;
  }
}
```

```
      int[] answer = {};

      if(arr.length-1 == 0) {
          answer = new int[1];

          return answer;

          }

                  //제일 작은 수 제거하기
```
